			}
		}
	}
	provinces_to_highlight = {
		owned_by = ROOT
	}
	selection_weight = {
		factor = 12
		modifier = {
			factor = 1.5
			tax_income_percentage = 0.3
		}
	}
	pre_effect = {
		random_owned_province = {
			limit = {
			}
			save_event_target_as = agenda_province
		}
	}
	task_requirements = {
		event_target:agenda_province = {
						 base_tax = 3
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_urban_patricians
			loyalty = 15
		}
		event_target:agenda_province = {
			add_province_modifier = {
				name = "patrician_development_bonus"
				duration = 5475
			}
		}
	}
	failing_effect = {
		add_estate_loyalty_modifier = {
			estate = estate_urban_patricians
			desc = EST_VAL_AGENDA_FAILED
			loyalty = -5
			duration = 7300
		}
	}
}
