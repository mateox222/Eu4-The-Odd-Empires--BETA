estate_rural_communes_develop_manpower = {
	can_select = {
		any_owned_province = {
			base_manpower = 3

		}
	}
	provinces_to_highlight = {
		owned_by = ROOT
		base_manpower = 3

	}
	selection_weight = {
		factor = 10
		modifier = {
			factor = 1.5
			manpower_percentage = 0.5
		}
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				base_manpower = 3

			}
			save_event_target_as = agenda_province
		}
	}
	task_requirements = {
		event_target:agenda_province = {
			add_base_manpower = 2
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_rural_communes
			loyalty = 15
		}
		add_yearly_manpower = 1.0
	}
	failing_effect = {
		add_estate_loyalty_modifier = {
			estate = estate_rural_communes
			desc = EST_VAL_AGENDA_FAILED
			loyalty = -5
			duration = 7300
		}
	}
}
