				AND = {
					religion = united_christianity

							value = 100

				AND = {
					NOT = { religion = united_christianity }
					OR = {
						NOT = { exists = PAP }
						PAP = {
							custom_trigger_tooltip = {
								tooltip = rivalled_or_rival_of
								OR = {
									is_rival = root
									root = { is_rival = prev }
								}
							}
							custom_trigger_tooltip = {
								tooltip = has_been_insulted_by_root
								OR = {
									has_opinion_modifier = {
										who = ROOT
										modifier = insulted
									}
									has_opinion_modifier = {
										who = ROOT
										modifier = scornfully_insulted
									}
								}
							}
						}
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					religion = united_christianity
				}
				add_reform_desire = -0.02
				add_papal_influence = 100
			}
			else = {
				add_dip_power = 50
			}
		}
	}
