# Events for New Institutions - EU4

namespace = new_institutions

# Urban Communes Institution
country_event = {
    id = new_institutions.1
    title = "new_institutions.1.t"
    desc = "new_institutions.1.d"
    picture = CITY_DEVELOPMENT_eventPicture
    
    is_triggered_only = yes

    goto = institution_origin
    	immediate = {
		hidden_effect = {
			from = { save_event_target_as = institution_origin }
		}
	}
    option = {
        name = "new_institutions.1.a"
 capital_scope = {
            add_institution_embracement = {
                which = urban_communes
                value = 100
            }
        }
        add_prestige = 10
        		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
    }
}
