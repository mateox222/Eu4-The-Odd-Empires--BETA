# The Rise of Platonic Philosophy
country_event = {
	id = religious_conflict.1
	title = religious_conflict.1.t
	desc = religious_conflict.1.d
	picture = RELIGION_eventPicture
	
	trigger = {
            has_global_flag = reformation_started
		religion = united_christianity
		any_owned_province = {
			has_building = university
		}
		NOT = { has_country_flag = platonic_movement_started }
	}
	
	mean_time_to_happen = {
		months = 120
		
		modifier = {
			factor = 0.8
			advisor = philosopher
		}
		modifier = {
			factor = 0.7
			innovativeness_ideas = 3
		}
		modifier = {
			factor = 1.2
			religious_unity = 0.8
		}
	}
	
	immediate = {
		set_country_flag = platonic_movement_started
	}
	
	option = {
		name = religious_conflict.1.a
		ai_chance = { factor = 60 }
		add_country_modifier = {
			name = "platonic_tolerance"
			duration = 3650
		}
        random_owned_province = {
    limit = { religion = united_christianity }
    change_religion = protestant
            }
		random_owned_province = {
			limit = { has_building = university }
			spawn_rebels = {
				type = religious_rebels
				size = 1
			}
		}
	}
	
	option = {
		name = religious_conflict.1.b
		ai_chance = { factor = 40 }
		add_stability = -1
		add_country_modifier = {
			name = "suppressed_philosophy"
			duration = 1825
		}
	}
}
