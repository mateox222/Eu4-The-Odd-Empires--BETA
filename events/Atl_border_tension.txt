# Border Skirmish
country_event = {
	id = border_tensions.3
	title = border_tensions.3.t
	desc = border_tensions.3.d
	picture = BORDER_TENSION_eventPicture
	
	trigger = {
		has_disaster = border_tensions
		any_neighbor_country = {
			is_enemy = ROOT
		}
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	option = {
		name = border_tensions.3.a # Retaliate
		random_neighbor_country = {
			limit = {
				is_enemy = ROOT
			}
			add_opinion = {
				who = ROOT
				modifier = border_skirmish
			}
		}
		add_army_tradition = 3
		add_mil_power = -25
	}
	
	option = {
		name = border_tensions.3.b # Ignore provocation
		add_prestige = -5
	}
}

			}

			}
		}
	}
}

# Fortification Demands
country_event = {
	id = border_tensions.5
	title = border_tensions.5.t
	desc = border_tensions.5.d
	picture = FORT_eventPicture
	
	trigger = {
		has_disaster = border_tensions
		any_owned_province = {
			any_neighbor_province = {
				NOT = { owned_by = ROOT }
				owner = {
					is_enemy = ROOT
				}
			}
			NOT = { has_building = fort_15th }
			NOT = { has_building = fort_16th }
		}
	}
	
	mean_time_to_happen = {
		months = 150
	}
	
	option = {
		name = border_tensions.5.a # Build fortifications
        random_list = {
	20 = {
		random_owned_province = {
			limit = {
				any_neighbor_province = {
					NOT = { owned_by = ROOT }
					owner = {
						is_enemy = ROOT
					}
				}
				NOT = { has_building = fort_15th }
			}
			add_building = fort_15th
		}
		add_years_of_income = -0.5
	}
    20 = { add_prestige = 5 }
    60 = { add_mil_power = 25 }

        }
    }
	
	option = {
		name = border_tensions.5.b # Rely on diplomacy
		add_dip_power = -50
		add_prestige = -5
	}
}
