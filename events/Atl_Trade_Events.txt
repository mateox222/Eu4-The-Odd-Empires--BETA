namespace = trade_empire

 
country_event = {
    id = trade_empire.1
    title = trade_empire.1.t
    desc = trade_empire.1.d
    picture = TRADEGOODS_eventPicture
    is_triggered_only = yes
    trigger = {
        government = republic
        mercantilism = 25
        num_of_merchants = 4
        calc_true_if = {
            all_trade_node = {
                trade_share = {
                    country = ROOT
                    share = 25
                }
            }
            amount = 4
        }
        NOT = { has_country_flag = golden_trade_age }
    }
    
 
    
    option = {
        name = trade_empire.1.a
        add_country_modifier = {
            name = golden_age_of_trade
            duration = 3650
        }
        add_prestige = 30
        add_mercantilism = 2
        set_country_flag = golden_trade_age
    }
    
    option = {
        name = trade_empire.1.b
        add_years_of_income = 1.5
        add_mercantilism = 5
        set_country_flag = golden_trade_age
    }
}
