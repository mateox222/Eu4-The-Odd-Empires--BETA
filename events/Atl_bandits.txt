namespace = bandit_plague

# Bandit Plague Begins
country_event = {
	id = bandit_plague.1
	title = bandit_plague.1.t
	desc = bandit_plague.1.d
	picture = phoenix_civil_war_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = bandits_suppressed
	}
	
	option = {
		name = bandit_plague.1.a
		add_stability = -1
		random_owned_province = {
			limit = {
				OR = {
					has_terrain = hills
					has_terrain = mountain
					has_terrain = forest
				}
			}
			add_unrest = 3
		}
	}
}

# Merchant Caravan Attack
country_event = {
	id = bandit_plague.3
	title = bandit_plague.3.t
	desc = bandit_plague.3.d
	picture = MERCHANTS_TALKING_eventPicture
	
	trigger = {
		has_disaster = bandit_plague
		any_owned_province = {
			OR = {
				has_building = marketplace
				has_building = trade_depot
			}
		}
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	option = {
		name = bandit_plague.3.a # Send military escort
		add_mil_power = -25
		add_army_tradition = 2
		random_owned_province = {
			limit = {
				has_building = marketplace
			}
			add_province_modifier = {
				name = military_escort
				duration = 1825
			}
		}
	}
	
	option = {
		name = bandit_plague.3.b # Pay ransom
		add_years_of_income = -0.3
		add_prestige = -5
	}
}

# Bandit Leader Emerges
country_event = {
	id = bandit_plague.5
	title = bandit_plague.5.t
	desc = bandit_plague.5.d
	picture = praetoriaguards_eventPicture
	
	trigger = {
		has_disaster = bandit_plague
		average_unrest = 3
	}
	
	mean_time_to_happen = {
		months = 150
	}
	
	option = {
		name = bandit_plague.5.a # Offer amnesty
		add_years_of_income = -0.2
		every_owned_province = { add_unrest = -2 }
		add_prestige = -10
	}
	
	option = {
		name = bandit_plague.5.b # Put bounty on his head
		add_years_of_income = -0.1
		set_country_flag = bandit_leaders_eliminated
		random_owned_province = {
			limit = {
				OR = {
					has_terrain = hills
					has_terrain = mountain
					has_terrain = forest
				}
			}
			spawn_rebels = {
				type = peasant_rebels
				size = 2
			}
		}
	}
}
