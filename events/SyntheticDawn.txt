namespace = synthetic_dawn

# Synthetics!
country_event = {
	id = synthetic_dawn.1
	title = synthetic_dawn.1.t
	desc = synthetic_dawn.1.d
	picture = SYNTHETICS_eventPicture

	trigger = {
		NOT = { tag = SYN }
		any_province = {
			owned_by = SYN
			has_discovered = ROOT
		}
		NOT = { has_country_flag = discovered_synthetics }
	}

	mean_time_to_happen = {
		months = 1
	}
	option = {
		name = "synthetic_dawn.1.a"
		set_country_flag = discovered_synthetics
		if = {
			limit = {
				NOT = {
					has_global_flag = synthetics_discovered_global
				}
			}
			set_global_flag = synthetics_discovered_global
		}
	}
}

# Province occupied by Synthetics
province_event = {
	id = synthetic_dawn.4
	title = synthetic_dawn.4.t
	desc = {
		trigger = {
			NOT = {
				province_id = 118
				owned_by = PAP
			}
		}
		desc = synthetic_dawn.4.d
	}
	desc = { # stand of the swiss guard
		trigger = {
			province_id = 118
			owned_by = PAP
		}
		desc = synthetic_dawn.4.d2
	}
	picture = SYNTHETICS_eventPicture

	is_triggered_only = yes

	trigger = {
		OR = {
			owner = {
				NOT = { tag = SYN }
			}
			owner = {
				NOT = { tag = XX1 }
			}
			owner = {
				NOT = { tag = XX2 }
			}
			owner = {
				NOT = { tag = XX3 }
			}
			owner = {
				NOT = { tag = XX4 }
			}
		}
		OR = {
			controller = {
				tag = SYN
			}
			controller = {
				tag = XX1
			}
			controller = {
				tag = XX2
			}
			controller = {
				tag = XX3
			}
			controller = {
				tag = XX4
			}
		}
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					is_subject_of = XX1
					NOT = { is_subject_of_type =autonomous_ai }
				}
				grant_independence = yes
			}

		}
		hidden_effect = {
			if = {
				limit = {	continent = north_america }
				cede_province = XX2
				add_core = XX2
			}
			if = {
				limit = {	continent = south_america }
				cede_province = XX2
				add_core = XX2
			}
			if = {
				limit = {	continent = europe }
				cede_province = XX4
				add_core = XX4
			}
			if = {
				limit = {	continent = asia }
				cede_province = XX3
				add_core = XX3
			}
			if = {
				limit = {
					NOT = {
						OR = {
							continent = asia
							continent = europe
							continent = south_america
							continent = north_america
						}
					}
				}

				cede_province = XX1
				add_core = XX1
			}

			set_base_tax = 1
			set_base_production = 1
			set_base_manpower = 1
			if = {
				limit = {
					province_id = 118
					owned_by = PAP
					PAP = {
						NOT = {
							has_country_modifier = swiss_guard_last_stand
						}
					}
				}
				PAP = {
					add_country_modifier = {
						name = "swiss_guard_last_stand"
						duration = -1
					}
				}
			}
		}
	}
	option = {
		name = "synthetic_dawn.4.a"
		tooltip = {
			cede_province = XX1
			add_core = XX1
			set_base_tax = 1
			set_base_production = 1
			set_base_manpower = 1
			if = {
				limit = {
					province_id = 118
					owned_by = PAP
					PAP = {
						NOT = {
							has_country_modifier = swiss_guard_last_stand
						}
					}
				}
				PAP = {
					add_country_modifier = {
						name = "swiss_guard_last_stand"
						duration = -1
					}
				}
			}
		}
	}
}

# Mayans open a temporal rift to fight the synthetics
country_event = {
	id = synthetic_dawn.5
	title = synthetic_dawn.5.t
	desc = synthetic_dawn.5.d
	picture = TEMPORAL_RIFT_eventPicture

	major = yes

	fire_only_once = yes

	trigger = {
		has_dlc = "Star and Crescent"
		NOT = { tag = SYN }
		has_country_flag = discovered_synthetics
		religion = mesoamerican_religion
	}
	mean_time_to_happen = {
		months = 6
	}
	option = {
		name = "synthetic_dawn.5.a"
		every_country = {
			limit = {
				religion = mesoamerican_religion
			}
			add_country_modifier = {
				name = mayan_dinosaur_cavalry
				duration = -1
			}
			add_country_modifier = {
				name = mayan_ufo_support
				duration = -1
			}
		}
	}
}
