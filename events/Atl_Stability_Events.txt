# Crisis of Authority
country_event = {
	id = low_stability.1
	title = low_stability.1.t
	desc = low_stability.1.d

	is_triggered_only = yes
	
	trigger = {
		NOT = { stability = 1 }
		NOT = { has_country_flag = had_authority_crisis_event }
		NOT = { legitimacy = 80 }
		NOT = { republican_tradition = 80 }
	}
	
	immediate = {
		set_country_flag = had_authority_crisis_event
	}
	
	option = {
		name = low_stability.1.a
		ai_chance = { factor = 40 }
		
		random_list = {
			40 = {
				add_country_modifier = {
					name = authority_restored
					duration = 3650
				}
				add_legitimacy = 10
				add_republican_tradition = 5
			}

				add_stability = -1

						type = pretender_rebels
						size = 1
					}
				}
			}
		}
	}
	
	option = {
		name = low_stability.1.b

			name = harsh_measures
			duration = 1825
		}

					name = popular_concessions
