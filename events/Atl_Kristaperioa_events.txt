# Druidic Renaissance
country_event = {
	id = flavor_kri.300
	title = flavor_kri.300.t
	desc = flavor_kri.300.d
	picture = BIG_BOOK_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = flavor_kri.300.a # Establish Great Library
		capital_scope = {
			add_permanent_province_modifier = {
				name = "kri_great_library"
				duration = -1
			}
		}
	}
	
	option = {
		name = flavor_kri.300.b # Create Druidic Schools
		every_owned_province = {
			limit = {
				religion = catholic
				development = 15
			}
			add_province_modifier = {
				name = "kri_druid_school"
				duration = -1
			}
		}
	}
	
	option = {
		name = flavor_kri.300.c # Send knowledge to colonies
		every_subject_country = {
			limit = { is_colonial_nation = yes }
			add_country_modifier = {
				name = "kri_colonial_wisdom"
				duration = 9125
			}
		}
	}
}

# The Atlantean Vision
country_event = {
	id = flavor_kri.400
	title = flavor_kri.400.t
	desc = flavor_kri.400.d
	picture = COLONIZATION_eventPicture
	
	trigger = {
		tag = A49
		has_ruler_modifier = kri_atlantean_blood
		num_of_colonies = 3

	}
	
	option = {
		name = flavor_kri.400.a # Follow the vision west
		add_country_modifier = {
			name = "kri_western_calling"
			duration = 7300

	}
	
	option = {
		name = flavor_kri.400.b # Focus on current lands

		set_country_flag = kri_atlantean_vision
	}
