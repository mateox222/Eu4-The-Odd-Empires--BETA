# Celtic Traditions in the New World
 country_event = {
	id = breton_colonial.1
	title = breton_colonial.1.t
	desc = breton_colonial.1.d
	picture = COLONIZATION_eventPicture
	
	trigger = {
		tag = BRI
		num_of_colonies = 3
		any_owned_province = {
			colonial_region = colonial_the_carribean
			culture = breton
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					colonial_region = colonial_the_carribean
					culture = breton
				}
				save_event_target_as = celtic_settlement
			}
		}
	}
	
	option = {
		name = breton_colonial.1.a
		ai_chance = { factor = 100 }
		
		event_target:celtic_settlement = {
			add_base_tax = 1
			add_base_production = 1
			add_province_modifier = {
				name = "bri_celtic_traditions"
				duration = 3650
			}
		}
		add_prestige = 10
	}
}

# Sugar Plantation Expansion
 country_event = {
	id = breton_colonial.3
	title = breton_colonial.3.t
	desc = breton_colonial.3.d
	picture = PLANTATION_eventPicture
	
	trigger = {
		tag = BRI
		has_country_modifier = bri_sugar_monopoly
		any_owned_province = {
			colonial_region = colonial_the_carribean
			trade_goods = sugar
		}
	}
	
	mean_time_to_happen = {
		months = 100
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					colonial_region = colonial_the_carribean
					trade_goods = sugar
				}
				save_event_target_as = sugar_plantation
			}
		}
	}
	
	option = {
		name = breton_colonial.3.a
		ai_chance = { factor = 60 }
		
		add_treasury = 200
		event_target:sugar_plantation = {
			add_base_production = 2
		}
	}
	
	option = {
		name = breton_colonial.3.b
		ai_chance = { factor = 40 }
		
		add_treasury = 100
		event_target:sugar_plantation = {
			add_base_production = 1
			add_base_tax = 1
			add_province_modifier = {
				name = "bri_model_plantation"
				duration = 5475
			}
		}
	}
}

# Colonial Scholars

	id = breton_colonial.4

	desc = breton_colonial.4.d
	picture = UNIVERSITY_eventPicture

		tag = BRI

				colonial_region = colonial_colombia
			}

		}
	}

	}

		name = breton_colonial.4.a

			limit = {
				OR = {
					colonial_region = colonial_the_carribean
					colonial_region = colonial_colombia
				}
				development = 10
			}
			add_building = university
		}
	}
}
