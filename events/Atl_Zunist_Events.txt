namespace = nepal_zunist

# The Sacred Flame Awakens
nepal_zunist.1 = {
	type = country_event
	placement = ROOT
	
	title = nepal_zunist.1.t
	desc = nepal_zunist.1.d
	picture = RELIGION_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			set_country_flag = npl_zunist_awakening
		}
	}
	
	trigger = {
		tag = NPL
		NOT = { religion = zunist }
		NOT = { has_country_flag = npl_zunist_awakening }
	}
	
	option = {
		name = nepal_zunist.1.a
		ai_chance = { factor = 80 }
		
		change_religion = zunist
		add_stability = -2
		add_country_modifier = {
			name = "npl_religious_upheaval"
			duration = 1825
		}
		random_owned_province = {
			limit = { is_capital = yes }
			change_religion = zunist
		}
		
		every_owned_province = {
			limit = { 
				NOT = { religion = zunist }
				NOT = { is_capital = yes }
			}
			add_local_autonomy = 25
		}
	}
	
	option = {
		name = nepal_zunist.1.b
		ai_chance = { factor = 20 }
		
		add_stability = 1
		add_legitimacy = 10
		add_country_modifier = {
			name = "npl_traditional_faith"
			duration = 3650
		}
	}
}
