# The Mountain Pilgrimage
country_event = {
	id = nepal_zunist.2
	title = nepal_zunist.2.t
	desc = nepal_zunist.2.d
	picture = RELIGION_eventPicture
	
	trigger = {
		tag = NPL
		religion = ibadi
		NOT = { has_country_flag = npl_mountain_pilgrimage }
		owns = 557
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = npl_mountain_pilgrimage
		}
	}
	
	option = {
		name = nepal_zunist.2.a
		ai_chance = { factor = 70 }
		
		add_adm_power = -50
		add_dip_power = -25
		add_fervor = 15
		557 = {
			add_base_manpower = 1
			add_permanent_province_modifier = {
				name = "npl_sacred_pilgrimage_site"
				duration = -1
			}
		}
	}
	
	option = {
		name = nepal_zunist.2.b
		ai_chance = { factor = 30 }
		
		add_adm_power = 25
		add_dip_power = 25
		add_mil_power = 25
	}
}
